# 가로 막대 차트
function drawBarChart(dataArray, targetId) {
    var data = google.visualization.arrayToDataTable(dataArray);
    var options = {
    legend: { position: 'none' }, // 범례 숨김
    colors: [colors.navyMid],
    hAxis: { minValue: 0 },       // 가로축 최소값 0
    animation: { startup: true, duration: 1000, easing: 'out' }
};
    // ColumnChart 대신 BarChart 사용
    var chart = new google.visualization.BarChart(document.getElementById(targetId));
    chart.draw(data, options);
}


# 누적 세로 막대 차트
function drawStackedColumnChart(dataArray, targetId) {
    var data = google.visualization.arrayToDataTable(dataArray);
    var options = {
    isStacked: true, // ★ 핵심: 막대를 쌓아 올리는 옵션
    legend: { position: 'top', maxLines: 2 },
    colors: [colors.error, colors.warning, colors.navy],
    vAxis: { minValue: 0 },
    animation: { startup: true, duration: 1000, easing: 'out' }
};
    var chart = new google.visualization.ColumnChart(document.getElementById(targetId));
    chart.draw(data, options);
}


# 영역 차트
function drawAreaChart(dataArray, targetId) {
    var data = google.visualization.arrayToDataTable(dataArray);
    var options = {
    legend: { position: 'bottom' },
    colors: [colors.navyLight, colors.navyMid],
    vAxis: { minValue: 0 },
    // 선 아래 영역의 투명도 조절
    areaOpacity: 0.3,
    animation: { startup: true, duration: 1000, easing: 'out' }
};
    var chart = new google.visualization.AreaChart(document.getElementById(targetId));
    chart.draw(data, options);
}



# 콤보 차트 (막대 + 선 혼합)
function drawComboChart(dataArray, targetId) {
    var data = google.visualization.arrayToDataTable(dataArray);
    var options = {
    vAxis: { title: '발생 건수' },
    hAxis: { title: '월' },
    seriesType: 'bars', // 기본은 모두 막대로 그림
    // 0번(첫번째) 데이터는 선 그래프로 변경 (오버라이드)
    series: { 0: { type: 'line', color: colors.error, curveType: 'function' } },
    colors: [colors.navyMid, colors.warning],
    animation: { startup: true, duration: 1000, easing: 'out' }
};
    var chart = new google.visualization.ComboChart(document.getElementById(targetId));
    chart.draw(data, options);
}


# 3D 파이 차트
function draw3DPieChart(dataArray, targetId) {
    var data = google.visualization.arrayToDataTable(dataArray);
    var options = {
    is3D: true, // ★ 핵심: 3D 효과 부여
    colors: [colors.navy, colors.navyMid, '#7DA0FA', '#A9C3FD'],
    chartArea: { width: '90%', height: '90%' },
    legend: { position: 'right' }
};
    var chart = new google.visualization.PieChart(document.getElementById(targetId));
    chart.draw(data, options);
}
